#!/usr/bin/env sh
"true" ; exec /usr/bin/env sbcl --noinform --script "$0" "$@"

(defpackage :DwightASpencerCom
  (:use #:cl)
  (:export #:finger #:Self))
(use-package :cl-user)

(defclass Self () ())
(defmethod AboutMe ((object Self))
  "{{ $a.bio | htmlEscape }}"
  `(:Resume
      (:Education   ({{ range $a.education }}{{ .symbol }} {{ end }}))
      (:Languages   ({{ range $a.languages }}{{ . }} {{ end }}))
      (:Projects    ({{ range $a.projects }}({{ .symbol }} "{{ .url }}") {{ end }}))
      (:Volunteering ({{ range $a.volunteering }}({{ .symbol }} "{{ .url }}") {{ end }}))
      (:Certs       ({{ range $a.certs }}({{ .symbol }} "{{ .date }}") {{ end }}))
      (:Tools       ({{ range $a.tools }}{{ . }} {{ end }}))))

(defun finger ()
  (let ((DwightSpencer (make-instance 'Self)))
    (format t "~a~&" (AboutMe DwightSpencer))))

(DwightASpencerCom:finger)
